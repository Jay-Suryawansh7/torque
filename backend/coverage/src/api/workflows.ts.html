type Handler = (req: Request, res: Response, next: NextFunction) => void | Promise<void>;
 
function wrap(fn: Handler): Handler {
  return (req, res, next) => {
    try {
      const result = fn(req, res, next);
      if (result instanceof Promise) result.catch(next);
    } catch (err) { next(err); }
  };
}
